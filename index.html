function binarray(a) {
  let count = 0;
  let countMap = new Map();
  countMap.set(0, -1);
  let maxLength = 0;
  
  for (let i = 0; i < a.length; i++) {
    if (a[i] === 1) {
      count++;
    } else {
      count--;
    }
    if (countMap.has(count)) {
      maxLength = Math.max(maxLength, 
                                 i - countMap.get(count));
    } else {
      countMap.set(count, i);
    }
  }
  return maxLength;
}
